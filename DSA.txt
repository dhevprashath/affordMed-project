Problem: Zigzag Conversion
--------------------------
The string "PAYPALISHIRING" is written in a zigzag pattern on a given number of rows like this:
P   A   H   N
A P L S I I G
Y   I   R
And then read line by line: "PAHNAPLSIIGYIR"

Signature: string convert(string s, int numRows);

Solution (C++):
--------------------------
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    string convert(string s, int numRows) {
        // Base case: If there's only 1 row or the string length is less than numRows,
        // the zigzag pattern is just the string itself.
        if (numRows == 1 || numRows >= s.length()) {
            return s;
        }

        // Create a vector of strings to represent each row of the zigzag pattern
        vector<string> rows(min(numRows, int(s.length())));
        int curRow = 0;
        bool goingDown = false;

        // Iterate through each character in the given string
        for (char c : s) {
            rows[curRow] += c;
            
            // Change direction when we hit the top row (0) or the bottom row (numRows - 1)
            if (curRow == 0 || curRow == numRows - 1) {
                goingDown = !goingDown;
            }
            
            // Move to the next row depending on the current direction
            curRow += goingDown ? 1 : -1;
        }

        // Combine all rows to construct the final zigzag string
        string result = "";
        for (string row : rows) {
            result += row;
        }
        
        return result;
    }
};

 * Complexity Analysis:
 * Time Complexity: O(n), where n == len(s). We visit every character exactly once.
 * Space Complexity: O(n), where n == len(s). The string builder/vector takes O(n) space.
